def forward(self,
    x: Tensor) -> Tensor:
  stem_bn = self.stem_bn
  stem_conv = self.stem_conv
  _0 = (stem_bn).forward((stem_conv).forward(x, ), )
  x0 = torch.gelu(_0)
  blocks = self.blocks
  x1 = (blocks).forward(x0, )
  global_pool = self.global_pool
  x2 = (global_pool).forward(x1, )
  x3 = torch.flatten(x2, 1)
  fc = self.fc
  return (fc).forward(x3, )
